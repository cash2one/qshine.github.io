简单记录下图的`深度优先遍历(DFS)`和`广度优先遍历(BFS)`

**注:** 一般深度优先采用递归方式, 广度优先采用队列方式

```python
# coding:utf-8

class Graph(object):
    def __init__(self):
        # 键为顶点, 值为列表是该顶点的邻居
        self.node_neighbors = {}
        self.visited = set()

    def nodes(self):
        return self.node_neighbors.keys()

    def add_nodes(self, nodeList):
        """添加所有顶点"""
        all_nodes = self.nodes()
        for node in nodeList:
            if node not in self.nodes():
                self.node_neighbors[node] = []

    def add_edge(self, edge):
        """添加边"""
        u, v = edge
        # 检查两个顶点是否在对方的邻居里面
        if (u not in self.node_neighbors[v]) and (v not in self.node_neighbors[u]):
            if u==v:
                self.node_neighbors[v].append(u)
            else:
                self.node_neighbors[u].append(v)
                self.node_neighbors[v].append(u)


    def dfs(self, root):
        """
        深度优先搜索
        :param root:
        :return:
        """
        if root not in self.visited:
            print(root)
            self.visited.add(root)
            for node in self.node_neighbors[root]:
                self.dfs(node)


    def bfs(self, root):
        """
        广度优先搜索
        :param root:
        :return:
        """
        queue = [root]
        while len(queue) != 0:
            node = queue.pop(0)
            print(node)
            self.visited.add(node)
            for i in self.node_neighbors[node]:
                if (i not in self.visited) and (i not in queue):
                    queue.append(i)
```
